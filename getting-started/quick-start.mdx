## Your First Component

Import and use a TARMAC component:

```tsx
import { Button } from '@tarmac/design-system';

function MyApp() {
  return (
    <Button variant="primary" onClick={() => alert('Hello TARMAC!')}>
      Click Me
    </Button>
  );
}
```

## Build a Login Form

Combine multiple components to create a functional form:

```tsx
import { Input, Button, Card } from '@tarmac/design-system';
import { useState } from 'react';

function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');

  return (
    <Card>
      <form onSubmit={(e) => { e.preventDefault(); }}>
        <Input
          type="email"
          label="Email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          required
        />
        <Input
          type="password"
          label="Password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          required
        />
        <Button type="submit" variant="primary" fullWidth>
          Sign In
        </Button>
      </form>
    </Card>
  );
}
```
